%============================================================
\emph{lbrace}, \emph{rbrace}, \emph{comma}, and \emph{number}
are terminal symbols (i.e., tokens).

\begin{eqnarray}
  S & \rightarrow & \mathit{lbrace} ~L~ \mathit{rbrace}
\\
  L & \rightarrow & L ~\mathit{comma}~ I
\\
  L & \rightarrow & I
\\
  I & \rightarrow & \mathit{number}
\\
  I & \rightarrow & S
\end{eqnarray}
